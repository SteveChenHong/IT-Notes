```
using System.Diagnostics;

using (var ct = new CodeTimer())
{
    Thread.Sleep(5000);
}

Console.ReadLine();

public class CodeTimer : IDisposable
{
    private Stopwatch stopwatch;

    public CodeTimer()
    {
        stopwatch = new Stopwatch();
        stopwatch.Start();
    }

    public void Dispose()
    {
        stopwatch.Stop();
        Console.WriteLine("執行時間：{0}", stopwatch.ElapsedMilliseconds);
    }
}
```
